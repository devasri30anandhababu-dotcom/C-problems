#include<stdio.h>
void find(int n){
    int sum=0;
    int num=n;
    for(int i=1;i<n;i++){
        if(n%i==0){
            sum+=i;
        }
    }
    
    if(sum==num){
        printf("Perfect Number");
    }
    else{
        printf("Not Perfect Number");
    }

}

int main(){
    int n;
    scanf("%d",&n);
    find(n);
    return 0;

}
