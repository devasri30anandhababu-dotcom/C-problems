#include <stdio.h>
void check(int n){
    int prime;
    for(int i=0;i<=n;i++){
        prime=0;
        for(int j=1;j<=i;j++){
            if(i%j==0){
                prime++;
            }
        }
    }
    if(prime==2){
            printf("Prime");
        }
        else{
            printf("Not Prime");
        }
}




int main() {
    int n;
    scanf("%d",&n);
    check(n);
    
 
    return 0;
}
