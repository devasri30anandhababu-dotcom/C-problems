#include<stdio.h>
void find(int a, int b, int operation){
    switch(operation){
        case 1:
        printf("%d",a+b);
        break;
        case 2:
        printf("%d",a-b);
        break;
        case 3:
        printf("%d",a*b);
        break;
        case 4:
        printf("%d",a/b);
        break;
        default:
        printf("Invalid");
        break;
    }
}

int main(){
    int a,b;
    scanf("%d%d",&a,&b);
    int operation;
    scanf("%d",&operation);
    find(a,b, operation);
    

    return 0;
}
