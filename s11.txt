#include<stdio.h>
int main(){
    int plan;
    scanf("%d",&plan);
    float data;
    scanf("%f",&data);
    switch(plan){
        case 1:
        if(data<=1.0){
            printf("Normal Speed");
        }
        else if(data>1.0){
            printf("Speed Reduced");
        }
        break;
        case 2:
        if(data <=2.0){
            printf("Normal speed");
        }
        break;
        case 3:
        if(data>2.0){
            printf("Extra Charges Applied");
        }
        break;
    }
    return 0;
}
