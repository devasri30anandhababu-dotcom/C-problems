#include <stdio.h>
int main(){
    char str[50];
    fgets(str,sizeof(str),stdin);
    int target;
    int index=-1;
    scanf("%c",&target);
    for(int i=0;str[i]!='\0';i++){
        if(target==str[i]){
            if(str[i]=='\n'){
                continue;
            }
            index=i;
            break;
        }
    }
    printf("%d",index);
    return 0;
}
