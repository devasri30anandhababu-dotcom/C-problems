#include <stdio.h>
int main(){
    char str[50];
    int i;
    fgets(str,sizeof(str),stdin);
    char str1[50];
    for(i=0;str[i]!='\0';i++){
        if(str[i]=='\n'){
            continue;
            
        }
        str1[i]=str[i];
        
    }
    str1[i]='\0';
    printf("%s",str1);
    return 0;
}
