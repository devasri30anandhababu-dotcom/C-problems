#include <stdio.h>
#include<string.h>

int main() {
    char str[100];
    scanf("%s",&str);
    int len=strlen(str);
    int palin=0;
    for(int i=0;i<len/2;i++){
        if(str[i]!=str[len-1-i]){
            palin=1;
            break;
        }
    }
    if(palin==0){
        printf("Palindrome");
    }
    else{
        printf("Not Palindrome");
    }
    return 0;
}
