#include<stdio.h>
int main(){
    char str[100];
    scanf("%s",str);
    char *p=str;
    int length=0;
    while(*(p+length)!='\0'){
        length++;
    }
    int start=0;
    int end=length-1;
    int palin=0;
    while(start<end){
        
        if(*(p+start)!=*(p+end)){
        palin=1;
        break;
    }
    start++;
    end--;
    }
    if(palin==0){
        printf("Yes");
    }
    else{
        printf("No");
    }
    return 0;
    
    
}
