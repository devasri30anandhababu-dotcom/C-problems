#include<stdio.h>
int main(){
    char str1[100];
    int i;
    fgets(str1,sizeof(str1),stdin);
    char str2[100];
    char *p=str1;
    char *p1=str2;
    for(i=0;*(p+i)!='\0';i++){
        if(*(p+i)=='\n'){
            continue;
        }
        *(p1+i)=*(p+i);
    }
    *(p1+i)='\n';
    printf("%s",(p1));
    return 0;
    
}
