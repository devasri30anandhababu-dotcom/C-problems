#include<stdio.h>
 int main(){
     int r,c;
     scanf("%d%d",&r,&c);
     int arr[r][c];
     for(int i=0;i<r;i++){
         for(int j=0;j<c;j++){
             scanf("%d",&arr[i][j]);
         }
     }
     int max=arr[0][0],min=arr[0][0],res,count=0;
     for(int i=0;i<r;i++){
         for(int j=0;j<c;j++){
            for(int k=i+1;k<r;k++){
             if(arr[i][j]>max){
                 max=arr[i][j];
                 if(arr[i][j]<min){
                 min=arr[i][j];
                 }
             res=max-min;
             if(res<=1000){
                 count++;
             
                 }
             }
              }
             
           
         }
     }
     printf("%d",count);
    
     return 0;
     
    }
