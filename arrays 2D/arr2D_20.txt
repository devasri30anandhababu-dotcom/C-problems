#include <stdio.h>

int main() {
    int r,c;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int repeat=0;
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            for(int k=i;k<r;k++){
                for(int l=j;l<c;l++){
                    if(k==i&&l==j){
                        continue;
                    }
                    if(arr[i][j]==arr[k][l]){
                        repeat=1;
                        break;
                    }
                }
            }
            if(repeat==0){
                printf("%d",arr[i][j]);
                return 0;
            }
        }
            }
    return 0;
}
