#include <stdio.h>

int main() {
    int r,c;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    for(int i=0;i<r;i++){
        int alter=1;
        for(int j=0;j<c-1;j++){
            if(arr[i][j]%2==arr[i][j+1]%2){
                alter=0;
                break;
                }
        }
        if(alter==1){
            printf("%d",i);
        }
    }
    return 0;
}
