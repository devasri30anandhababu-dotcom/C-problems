#include<stdio.h>
int main(){
    int r,c,index=0;
    scanf("%d%d",&r,&c);
    int a[r][c];
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            scanf("%d",&a[i][j]);
        }
    }
    int min,avg;
    
    for(int i=0;i<r;i++){
        int total=0;
        int count=0;
        for(int j=0;j<c;j++){
            total=total+a[i][j];
            count++;
        }
            avg=total/count;
            if(avg<min){
                min=avg;
                index=i;
            }
            
        }
         printf("%d",index);
       
}
