#include <stdio.h>

int main() {
    int r,c;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int high=0;
    int second=-1;
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            if(arr[i][j]>high){
                second=high;
                high=arr[i][j];
            }
            else if(arr[i][j]>second && high!=arr[i][j]){
                second=arr[i][j];
            }
        }
    }
    printf("%d",second);
    return 0;
}
