 #include<stdio.h>
 int main(){
     int r,c;
     scanf("%d%d",&r,&c);
     int a[r][c];
     for(int i=0;i<r;i++){
         for(int j=0;j<c;j++){
             scanf("%d",&a[i][j]);
         }
     }
     int count;
     for(int i=0;i<r;i++){
         for(int j=0;j<c;j++){
             for(int k=0;k<j;k++){
                 if(a[i][j]%a[k][j]==0){
                     count++;
                 }
             }
            
         }
        
     }
    printf("%d",count);
     
     
    return 0; 
 }
