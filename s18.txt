#include<stdio.h>
int main(){
    int attempts;
    scanf("%d",&attempts);
    int marks;
    scanf("%d",&marks);
    switch(attempts){
        case 1:
        if(marks>=80){
            printf("Excellent");
        }
        else{
            printf("Not Qualified");
        }
        break;
        case 2:
        if(marks >=60){
            printf("Good");
        }
        else{
            printf("Not Qualified");
        }
        break;
        default:
        printf("Needs Improvement");
    }
    return 0;
}
