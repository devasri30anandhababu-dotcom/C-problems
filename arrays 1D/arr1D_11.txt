#include<stdio.h>
int main(){
    int size,i;
    scanf("%d",&size);
    int arr[size];
    int total=0;
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
         total+=arr[i];
    }
    int left=0;
    int result=-1;
    
    for(int i=0;i<size;i++){
        int right=total-left-arr[i];
        if(right==left){
            result=i;
            break;
        }
         left+=arr[i];
    }
   
    printf("%d",result);
    
    return 0;
}
