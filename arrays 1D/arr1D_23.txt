#include <stdio.h>
int main() {
    int p,i;
    scanf("%d",&p);
    int arr[p];
    for(int i=0;i<p;i++){
        scanf("%d",&arr[i]);
    }
    int min=0,min2=-1;
    for(int i=0;i<p;i++){
        if(arr[i]>0){
            if(min==0 || arr[i]<min){
                min=arr[i];
            }
            else if(min2==-1 || arr[i]>min){
                min2=arr[i-1];
            }
        }
    }
        if(min!=0){
            printf("%d",min);
        }
        else{
            printf("%d",min2);
        }
    }
