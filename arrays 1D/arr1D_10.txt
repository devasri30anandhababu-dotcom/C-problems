#include <stdio.h>
int main() {
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int a,max=-1;
    for(int i=n-1;i>=0;i--){
        a=arr[i];
        arr[i]=max;
        if(a>max){
            max=a;
        }
    }
    for(int i=0;i<n;i++){
        printf("%d ",arr[i]);
    }
 

    return 0;
}
