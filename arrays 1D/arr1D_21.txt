#include <stdio.h>

int main() {
    int n,i,j,dup=0;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(arr[i]==arr[j]){
                printf("%d",arr[i]);
                dup=1;
                break;
    }
        }
        if(dup==1){
           break;
        }
        }
        if(dup==0){
            printf("-1");
        }
    

    return 0;
}
