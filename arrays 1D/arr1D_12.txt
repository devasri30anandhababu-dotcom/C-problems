#include<stdio.h>
int main(){
    int n,i,j;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int target,val=0;
    scanf("%d",&target);
    for(int i=1;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(arr[i]+arr[j]!=target){
                val=1;
            }
            }
           }
            if(val==0){
                printf("%d%d",arr[i],arr[j]);
            }
            else{
                printf("No pair");
            }
    return 0;
        
    }
