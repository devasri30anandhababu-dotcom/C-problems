
#include <stdio.h>

int main() {
    int n,i,j;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    for(int i=0;i<n;i++){
        int dup=0;
        for(int j=0;j<i;j++){
            if(arr[i]==arr[j]){
            dup=1;
            break;
    }
        }
        if(dup==0){
            printf("%d ",arr[i]);
            
        }
    }

    return 0;
}
