#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int min_price=arr[0];
    int max_profit=arr[1]-arr[0];
    for(int i=0;i<n;i++){
        int profit=arr[i]-min_price;
        if(profit>max_profit){
            max_profit=profit;
        }
        if(arr[i]<min_price){
            min_price=arr[i];
        }
    }
    printf("%d",max_profit);
    return 0;
    
}
