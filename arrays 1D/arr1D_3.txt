#include<stdio.h>
int main(){
    int size;
    scanf("%d",&size);
    int arr[size];
    for(int i=0;i<size;i++){
       scanf("%d",&arr[i]);
    }
    int isSorted=1;
    for(int j=0;j<size-1;j++){
        if(arr[j]>arr[j+1]){
            isSorted=0;
            break;
        }
    
    }
    if(isSorted){
        printf("Yes");
    }
    else{
        printf("No");
    }
    return 0;
}
