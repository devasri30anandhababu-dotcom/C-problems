#include <stdio.h>

int main() {
    int size;
    scanf("%d",&size);
    int arr[size-1];
    for(int i=0;i<size-1;i++){
        scanf("%d",&arr[i]);
    }
    int expected_sum=0;
    int actual_sum;
    for(int j=0;j<size-1;j++){
        expected_sum+=arr[j];
    }
    actual_sum=(size)*(size+1)/2;
    printf("Missing roll number : %d",actual_sum-expected_sum);
    
    return 0;
}
