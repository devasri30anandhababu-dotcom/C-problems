#include <stdio.h>

int main() {
    int marks,total=0,i;
    scanf("%d",&marks);
    int arr[marks];
    for(int i=0;i<marks;i++){
        scanf("%d",&arr[i]);
        total+=arr[i];
    }
    int avg,res,count=0;
    for(int i=0;i<marks;i++){
        avg=total/marks;
        if(avg<arr[i]){
            count++;
        }
    }
    printf("%d",count);


    return 0;
}
