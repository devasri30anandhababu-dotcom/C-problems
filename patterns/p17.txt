#include<stdio.h>
int main(){
    int i,j;
    char ch='A';
    for(int i=1;i<=4;i++){
        for(int k=1;k<=4-i;k++){
            printf(" ");
        
    }
    for(int j=1;j<i;j++){
        printf("%c",ch);
        ch++;
  
    }
    for(int j=i;j>0;j--){
        printf("%c",ch);
        ch--;
       
        
    }
    ch++;
    printf("\n");
    }
    return 0;
}
